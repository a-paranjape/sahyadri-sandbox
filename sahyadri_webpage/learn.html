Getting Started

# Quick-start example: Reading a Sahyadri halo catalog import numpy as np from sahyadri_tools import read_catalog, read_snapshot # Load halo catalog at z=0 for fiducial cosmology halos = read_catalog("fiducial", snap=100) # Access halo properties mass = halos["mvir"] # Virial mass [h-1 Msun] pos = halos["pos"] # Position [h-1 Mpc] vel = halos["vel"] # Velocity [km/s] # Filter resolved halos (>=40 particles) mask = mass >= 3.2e9 print(f"Resolved halos: {mass[mask].size}")
